E - 受欢迎的牛
思路：
有向图缩点，找出度为0的点集，点集大小即为答案，并且点集合个数为1
代码：

                  
    #include <bits/stdc++.h>
    using namespace std; 
    const int inf=0x3f3f3f3f;
    const long long INF=0x3f3f3f3f3f3f3f3f;
    const int maxn=2e5+50;
    const int mod=998244353;
    const int Mod=1e9+7;
    typedef long long ll;

    int n,m,idx,cnt;
    stack <int> stk;
    vector <int> g[maxn];
    vector <vector<int>> scc;
    int dfn[maxn],low[maxn],ins[maxn],bel[maxn]; 
    void tarjan(int x)
    {
      dfn[x]=low[x]=++idx;
      stk.push(x);
      ins[x]=1;
      for (auto y:g[x])
      {
        if (!dfn[y]) tarjan(y);
        if (ins[y]) low[x]=min(low[x],low[y]);
      }
      if (dfn[x]==low[x]){
        vector <int> c;
        ++cnt;
        while (true) {
          int y=stk.top();
          c.push_back(y);
          stk.pop(); 
          ins[y]=false;
          bel[y]=cnt;
          if (x==y) break;
        }
        scc.push_back(c);
      }
    }
    vector <int> G[maxn];
    int out[maxn];
    int main()
    {
      std::ios::sync_with_stdio(false),cin.tie(0),cout.tie(0);
      cin >> n >> m;
      
      map <pair<int,int>,int> mp;
      
      for (int i=1; i<=m; i++)
      {
        int x,y;
        cin >> x >> y;
        if (!mp[{x,y}]){ g[x].push_back(y); mp[{x,y}]=1; }
      }
      for (int i=1; i<=n; i++) if (!dfn[i]) tarjan(i);
      mp.clear();
      
      for (int i=1; i<=n; i++)
      {
        int x=bel[i];
        for (int j:g[i])
        {
          int y=bel[j];
          if (x==y) continue;
          if (!mp[{x,y}]) 
          {
            G[x].push_back(y);
            mp[{x,y}]=1;
            out[x]++;
            
          }
        }
      }
      ll ans=0,f=0;
      for (int i=1; i<=cnt; i++) 
      {
        if (out[i]==0) { ans+=scc[i-1].size(); f++;}
      }
      if (f!=1) cout << "0\n";
      else cout << ans << "\n";
    }

    }g[maxn];

    void add(int x,int y)

      ++etot;

      g[etot].nxt=head[x];

    void tarjan(int x)
    {
      dfn[x]=low[x]=++idx;
      sta[++top]=x;
      for(int i=head[x];i;i=g[i].nxt)
      {
        int y=g[i].y;
        if (!dfn[y]) 
        {
          tarjan(y);
          low[x]=min(low[x],low[y]);
        } 
        else 	if (!scc[y]) low[x]=min(low[x],low[y]);
        
      }
      if (dfn[x]==low[x])
      {
        sccnt++;
        while(sta[top]!=x)
        {
          scc[sta[top]]=sccnt;
          top--;
        }
        scc[sta[top]]=sccnt;
        top--;	
      }
    }

    int ans=0;
    int v[maxn];
    int dis[maxn],du[maxn];
    int val[maxn];
    int st,tot;
    set <pair<int,int>> qc;

    {
      std::ios::sync_with_stdio(false),cin.tie(0),cout.tie(0);
      cin >> n >> m;
      
      for (int i=1; i<=m; i++)
      {
        int x,y;
        cin >> x >> y;
        e.push_back({x,y});
        add(x,y);
        
      }
      
      for (int i=1; i<=n; i++) cin >> val[i];
      
      cin >> st >> tot;
      tarjan(st);
      
      for (int i=1; i<=n; i++){
        g[i].y=0; g[i].z=0;
      }
      for (int i=0; i<=etot+2; i++) head[i]=0;
      etot=0;
      
      for (int i=0; i<e.size(); i++)
      {
        int x=e[i].first,y=e[i].second;
        x=scc[x]; y=scc[y];
        if (x==y || x==0 || y==0) continue;
        if (qc.count({x,y})==0)
        {
          add(x,y);
          du[y]++;
          qc.insert({x,y});
        }
      }	
      for (int i=1; i<=n; i++) v[scc[i]]+=val[i];		
      
      st=scc[st];
      queue <int> q;
      q.push(st); 
      while (q.size())
      {
        ll x=q.front();
        q.pop();
        dis[x]+=v[x];
        for(int i=head[x];i;i=g[i].nxt)
        {
          int y=g[i].y;
          du[y]--;
          dis[y]=max(dis[y],dis[x]);
          if (du[y]==0) q.push(y);
        }
      }	
      for (int i=1; i<=tot; i++) 
      { 
        int id; cin >> id; 
        ans=max(ans,dis[scc[id]]);
      }
      cout << ans << "\n";
    }
G - 奶牛议会
思路：
2-sat,并且dfs判断方案
代码：

                  
    #include<bits/stdc++.h>
    using namespace std;
    const int maxn=3e5+50;
    int n,m;
    int head_size=0;
    struct edge{
      int x,y,z,nxt;
    }e[maxn];
    int head[maxn];
    int dfncnt=0,sccnt=0,top=0,cnt=0; 
    int dfn[maxn],low[maxn],scc[maxn],sta[maxn],si[maxn];
    void add(int x,int y)
    {
      cerr << x << " " << y << "\n";
      e[++head_size].x=x;
      e[head_size].y=y;
      e[head_size].nxt=head[x];
      head[x]=head_size;
    }
    void tarjan(int f) 
    {
      low[f]=dfn[f]=++dfncnt;
      sta[++top]=f;
      for(int i=head[f];i;i=e[i].nxt)
      {
        int t=e[i].y;
        if(!dfn[t])
        {
          tarjan(t);
          low[f]=min(low[f],low[t]);
        }
        else if(!scc[t]) low[f]=min(low[f],low[t]);
      }
      if(dfn[f]==low[f])
      {
        sccnt++;	
        si[++cnt]++;
        while(sta[top]!=f)
        {
          scc[sta[top]]=sccnt;
          top--;
          si[cnt]++;
        }
        scc[sta[top]]=sccnt;
        top--;
      }
    }
    void init(int n)
    {
      for (int i=0; i<=(n+1)*2; i++)
      {
        e[i].x=0; e[i].y=0;	e[i].nxt=0;
        head[i]=0;
        dfn[i]=0; low[i]=0;
        scc[i]=0; sta[i]=0;	si[i]=0;
      }
      head_size=0;
      dfncnt=0; sccnt=0; top=0; cnt=0; 
    }
    map<pair<int,int>,int> mpf;
    bool vis[maxn];
    bool dfs(int x,int y)
    {
      vis[x]=1;	
      if (x==y) return 1;
      for(int i=head[x];i;i=e[i].nxt)
      {
        int t=e[i].y;
        if (!vis[t]) 
          if (dfs(t,y)) return 1;
      }
      return 0;
    }
    bool check(int x,int y)
    {
      for (int i=0; i<2*n; i++) vis[i]=0;
      dfs(x,y);
      return vis[y];
    }
    void solve()
    {
      cin >> n >> m; 
      init(n);
      for(int i=1;i<=m;i++) 
      {
        char sx,sy;
        int x,y;
        cin >> x >> sx >> y >> sy;
        x--; y--;
        x=x*2+(sx=='Y');
        y=y*2+(sy=='Y');
        add(x^1,y);
        add(y^1,x);
      }
      for(int i=0;i<2*n;i++) if(!dfn[i]) tarjan(i);
      long long cntans=0;
      for (int i=0; i<n; i++)
        if (scc[i*2]==scc[i*2+1]) { cout << "IMPOSSIBLE\n"; return ;}
      for (int i=0; i<n; i++) 
      { 
        if (check(2*i,2*i+1)) cout << "Y";
        else if (check(2*i+1,2*i)) cout << "N";
        else cout << "?";
      }
    }
    int main()
    {
      int T=1;
      while (T--)
      {
        solve();
      }
    }
H - Redundant Paths G
思路：
边双缩点，得到一颗树，答案为叶子数+1/2
代码：

                  
    #include<bits/stdc++.h>
    using namespace std;
    const int maxn=1e6+5;
    int head[maxn],ver[maxn*2],Next[maxn],fro[maxn];
    int dfn[maxn],low[maxn];
    bool bridge[maxn];
    int n,m,tot=1,num; 
    int c[maxn],dcc;
    void dfs(int x)                           
    {
        c[x]=dcc;
        for(int i=head[x];i;i=Next[i])
        {
            int y=ver[i];
            if(c[y]||bridge[i]) continue;
            dfs(y);
        }
    }
    
    void add(int x,int y)
    {
        ver[++tot]=y,Next[tot]=head[x],head[x]=tot;
        fro[tot]=x;
    }
    long long du[maxn];
    void tarjan(int x,int in_edge)
    {
        dfn[x]=low[x]=++num;
        for(int i=head[x];i;i=Next[i])
        {
            int y=ver[i];
            if(!dfn[y])
            {
                tarjan(y,i);
                low[x]=min(low[x],low[y]);
                if(low[y]>dfn[x])
                {
                    bridge[i]=bridge[i^1]=true;
                }
            }
            else if(i!=(in_edge^1))
                low[x]=min(low[x],low[y]);
        }
    }

    vector <pair<int,int>> edge2;
    int main()
    {
      cin >> n >> m;
      for(int i=1;i<=m;i++) 
      {
        long long x,y;
        cin >> x >>y;
        add(x,y);
        add(y,x);
      }
      for(int i=1;i<=n;i++)
        if(!dfn[i]) tarjan(i,0);
      for (int i=1; i<=n; i++)
      {
        if (!c[i])
        {
          ++dcc;
          dfs(i);
        }
      }
      for(int i=1;i<=tot;i++)
      {
        int u=c[ver[i]],v=c[fro[i]]; 
        if(u!=v) edge2.push_back({u,v});
      }
      
      sort(edge2.begin(),edge2.end());
      edge2.erase(unique(edge2.begin(),edge2.end()),edge2.end());
      for (auto it:edge2)
      {
        du[it.first]++; du[it.second]++;
      }	
      long long cnt=0;
      for (int i=1; i<=dcc; i++) if (du[i]==2) cnt++;
      cout << (1+cnt)/2;
      return 0;
    }
I - BLO-Blockade
思路：
割点，树形dp
树形dp不会建议拷打yjf，他的dp课前，我要求让他讲了，虽然他当时状态萎靡
代码：

                  
    #include <bits/stdc++.h>
    using namespace std; 
    const int Mod=1e9+7;
    const int mod=998244353;
    const int inf=0x3f3f3f3f;
    const long long INF=0x3f3f3f3f3f3f3f3f;
    const int maxn=1e5+5;
    const int maxm=2e6+5;
    typedef long long ll;
    int n,m;
    int etot=1;
    struct edge{
      int y,nxt;
    }g[maxm*2];
    int head[maxn];
    void add(int x,int y)
    {
      ++etot;
      g[etot].y=y;
      g[etot].nxt=head[x];
      head[x]=etot;
    }

    int idx=0,cutot;
    int dfn[maxn],low[maxn];
    ll sz[maxn],ans[maxn];
    void tarjan(int x,int fa)
    {
      int child=0;
      dfn[x]=low[x]=++idx;
      sz[x]=1;
      ans[x]=n-1;
      ll cut=n-1;
      for (int i=head[x]; i; i=g[i].nxt)
      {
        int y=g[i].y;
        if (!dfn[y])
        {
          tarjan(y,x);
          sz[x]+=sz[y];
          low[x]=min(low[x],low[y]);
          if (low[y]>=dfn[x]){
            ans[x]+=sz[y]*(n-sz[y]);
            cut-=sz[y];
          } 
          if (x==fa) child++;
        }
        else low[x]=min(low[x],dfn[y]);
      }
      ans[x]+=cut*(n-cut);
    }

    map<pair<int,int>,bool> qc;
    int main()
    {
      std::ios::sync_with_stdio(false),cin.tie(0),cout.tie(0);
      cin >> n >> m;
      for (int i=1; i<=m; i++)
      {
        int x,y;
        cin >> x >> y;
        if (qc[{x,y}]==0) { add(x,y); qc[{x,y}]=1; }
        if (qc[{y,x}]==0) { add(y,x); qc[{y,x}]=1; }
      }
        cutot=0; 
      for (int i=1; i<=n; i++) if (!dfn[i]) tarjan(i,i);
      for (int i=1; i<=n; i++) cout << ans[i] << "\n";
    }
J - 所驼门王的宝藏
思路：
有向图缩点，离散化

    ll ans;
    int n,r,c;
    map <int,int> cid,rid;
    map <pair<int,int>,int> id;
    ll x[maxn],y[maxn],z[maxn],dp[maxn];
    vector <int> vc[maxn],vr[maxn];

    int idx=0,sccnt=0;
    vector <pair<int,int>> e;

      int y,nxt;

    void tarjan(int x)
    {
      dfn[x]=low[x]=++idx;
      sta[++top]=x;
      for(int i=head[x];i;i=g[i].nxt)
      {
        int y=g[i].y;
        if (!dfn[y]) 
        {
          tarjan(y);
          low[x]=min(low[x],low[y]);

      if (dfn[x]==low[x])
      {
        sccnt++;
        int sz=0;
        dp[sccnt]=0;
        while(1)
        {
          int y=sta[top];
          scc[y]=sccnt;
          sz+=(y<=n);
          for(int i=head[y];i;i=g[i].nxt)
          {
            int z=g[i].y;
            if (scc[z]!=sccnt && scc[z]!=0) 
              dp[sccnt]=max(dp[sccnt],dp[scc[z]]);
          }
          top--;
          if (y==x) break;
        }
        dp[sccnt]+=sz;
        ans=max(ans,dp[sccnt]);
      }
    }

小更新，更新了博客更新方式，从服务器下载docker变成了通过github的action自动对比更新
change.yml代码如下 
name: CI/CD Pipeline

on:
  push:
    branches:
      - '**' # This will match all branches

jobs:
  deploy:
    runs-on: ubuntu-latest

    steps:
    - name: Checkout code
      uses: actions/checkout@v2

    - name: Install sshpass
      run: sudo apt-get install sshpass

    - name: Create known_hosts
      run: |
        mkdir -p ~/.ssh
        touch ~/.ssh/known_hosts

    - name: Add server to known hosts
      run: ssh-keyscan IP >> ~/.ssh/known_hosts

    - name: Deploy to server
      run: |
        sshpass -p ${{ secrets.SERVER_SSH_PRIVATE_KEY }} rsync -r . root@IP:/tmp/html
        sshpass -p ${{ secrets.SERVER_SSH_PRIVATE_KEY }} ssh root@IP "docker cp /tmp/html blog:/usr/share/nginx && docker restart blog"
